## 一、项目概述

基于学长的《GIS工程应用实践》课程设计报告整理，本项目是一个**应急物资调度系统**，包含五大核心模块：

1. 三维洪水模拟
2. 应急物资调度模拟（二维）
3. 仓库物资管理
4. 卫星云图
5. 3D应急物资调度模拟

// 初始化Cesium视图
var viewer = new Cesium.Viewer('cesiumContainer', {
    terrainProvider: Cesium.createWorldTerrain(),  // STK地形
    imageryProvider: new Cesium.BingMapsImageryProvider({
        key: 'YOUR_BING_KEY'
    })
});

var hypFlood = new Cesium.HypsometricSetting();
var isPaused = false;
var savedEntities_buildings = [];
var savedEntities_roads = [];

// 洪水淹没函数
function floodUpdate(positions, floodHeight, floodSpeed) {
    hypFlood.CoverageArea = positions;
    hypFlood.MinVisibleValue = 0;
    hypFlood.MaxVisibleValue = floodHeight;
    
    viewer.scene.globe.HypsometricSetting = {
        hypsometricSetting: hypFlood,
        analysisMode: Cesium.HypsometricSettingEnum.AnalysisRegionMode.ARM_REGION
    };
    
    let startTime = viewer.clock.currentTime;
    let endTime = Cesium.JulianDate.addSeconds(
        startTime, 
        floodHeight / floodSpeed, 
        new Cesium.JulianDate()
    );
    
    viewer.clock.shouldAnimate = true;
    viewer.clock.startTime = startTime.clone();
    viewer.clock.stopTime = endTime.clone();
    viewer.clock.currentTime = startTime.clone();
    viewer.clock.clockRange = Cesium.ClockRange.CLAMPED;
    
    function updateFlood() {
        if (!isPaused) {
            let currentTime = viewer.clock.currentTime;
            let seconds = Cesium.JulianDate.secondsDifference(currentTime, startTime);
            let currentHeight = Math.min(seconds * floodSpeed, floodHeight);
            hypFlood.MaxVisibleValue = currentHeight;
            
            viewer.scene.globe.HypsometricSetting = {
                hypsometricSetting: hypFlood,
                analysisMode: Cesium.HypsometricSettingEnum.AnalysisRegionMode.ARM_REGION
            };
            
            document.getElementById('simulatedTime').textContent = seconds.toFixed(2);
            
            if (currentHeight >= floodHeight) {
                viewer.clock.onTick.removeEventListener(updateFlood);
                viewer.clock.shouldAnimate = false;
                // 统计淹没数据
                document.getElementById('houseCount').textContent = countSubmerged(savedEntities_buildings);
                document.getElementById('roadCount').textContent = countSubmerged(savedEntities_roads);
            }
        }
    }
    
    viewer.clock.onTick.addEventListener(updateFlood);
}

// 加载GeoJSON建筑数据
function loadGeoJSON1(url, color) {
    Cesium.Resource.fetchJson(url)
    .then(function (geojsonData) {
        var dataSource = Cesium.GeoJsonDataSource.load(geojsonData, {
            clampToGround: true
        });
        viewer.dataSources.add(dataSource)
        .then(function (dataSource) {
            var entities = dataSource.entities.values;
            savedEntities_buildings = entities;
            for (var i = 0; i < entities.length; i++) {
                var entity = entities[i];
                entity.polygon.material = color;
            }
        });
    })
    .otherwise(function (error) {
        console.error('加载GeoJSON数据失败:', error);
    });
}

// 统计淹没范围内的实体
function countSubmerged(entities) {
    var count = 0;
    var west, south, east, north; // 淹没区域边界
    for (var i = 0; i < entities.length; i++) {
        var entity = entities[i];
        var hierarchy = entity.polygon.hierarchy.getValue(Cesium.JulianDate.now());
        var positions = hierarchy.positions;
        if (positions.length > 0) {
            var cartesian = positions[0];
            var cartographic = Cesium.Cartographic.fromCartesian(cartesian);
            if (isPointInRectangle(cartographic, west, south, east, north)) {
                count++;
            }
        }
    }
    return count;
}

// 标记受灾人员位置
function addRescueMarker() {
    viewer.screenSpaceEventHandler.setInputAction(function (click) {
        const cartesian = viewer.scene.pickPosition(click.position);
        if (Cesium.defined(cartesian)) {
            const cartographic = Cesium.Cartographic.fromCartesian(cartesian);
            const longitude = Cesium.Math.toDegrees(cartographic.longitude);
            const latitude = Cesium.Math.toDegrees(cartographic.latitude);
            const height = cartographic.height;
            
            viewer.entities.add({
                position: Cesium.Cartesian3.fromDegrees(longitude, latitude, height),
                billboard: {
                    image: '../../map/position.png',
                    width: 30,
                    height: 40,
                    eyeOffset: new Cesium.Cartesian3(0.0, 0.0, -55.0)
                },
                name: "受灾人员标记",
                description: "需要救援"
            });
        }
    }, Cesium.ScreenSpaceEventType.LEFT_CLICK);
}

    if (points.length != 0) {

        polygon = L.polygon(points, {
            color: '#fc6a00',

        var geometryParam = new L.supermap.GetFeaturesByGeometryParameters({

        new L.supermap.FeatureService(url)
            .getFeaturesByGeometry(geometryParam)
            .then(function(serviceResult) {
                var result1 = serviceResult.result;
                var finalR = handleAvoidPoints(result1);
                bestWay(finalR, startP, endP, color);
            });
    } else {

// 最佳路径规划
function bestWay(avoidNodes, startP, endP, color) {
    var analystParameter;
    if (avoidNodes.length == 0) {
        analystParameter = new L.supermap.TransportationAnalystParameter({
            resultSetting: resultSetting,
            weightFieldName: "Length"
        });
    } else {
        analystParameter = new L.supermap.TransportationAnalystParameter({
            resultSetting: resultSetting,
            weightFieldName: "Length",
            barrierNodeIDs: avoidNodes
        });
    }
    
    var resultSetting = new L.supermap.TransportationAnalystResultSetting({
        returnEdgeFeatures: true,
        returnEdgeGeometry: true,
        returnEdgeIDs: true,
        returnNodeFeatures: true,
        returnNodeGeometry: true,
        returnNodeIDs: true,
        returnPathGuides: true,
        returnRoutes: true
    });
    
    var findPathParams = new L.supermap.FindPathParameters({
        isAnalyzeById: false,
        nodes: [L.latLng(startP[0], startP[1]), L.latLng(endP[0], endP[1])],
        hasLeastEdgeCount: false,
        parameter: analystParameter
    });
    
    var serviceUrl = "http://localhost:8090/iserver/services/transportationAnalyst-web4_0/rest/networkanalyst/web4_0_Network@web4_0";
    var findPathService = new L.supermap.NetworkAnalystService(serviceUrl);
    
    findPathService.findPath(findPathParams).then(function(serviceResult) {
        var result = serviceResult.result;
        handleline(result, color);
    });
}

// 路线解析与呈现
function handleline(res, color) {
    var features = res.pathList[0].pathGuideItems.features;
    var temp = [];
    for (var i = 0; i < features.length; i++) {
        var coordinate = features[i].geometry.coordinates;
        for (var j = 0; j < coordinate.length; j++) {
            if (typeof(coordinate[j]) == 'number') {
                temp.push([features[i].geometry.coordinates[1], features[i].geometry.coordinates[0]]);
                j++;
            } else {
                temp.push([coordinate[j][1], coordinate[j][0]]);
            }
        }
    }
    
    // 保存路径数据
    var content = JSON.stringify({path: temp.concat()});
    var blob = new Blob([content], {type: "text/plain;charset=utf-8"});
    saveAs(blob, "save.json");
    
    // 绘制路线
    L.polyline(temp.concat(), {
        color: color,
        fillColor: color,
        fillOpacity: 1
    }).addTo(map);
}

### 模块3：基于中值模型的应急物流选址算法（K-Means）

#### 5.3.1 功能描述

- 输入受灾点坐标
- 使用K-Means聚类确定最优配送中心位置
- 返回聚类中心作为临时物资点

#### 5.3.2 Python后端代码

```python
from flask import Flask, request, jsonify
from flask_cors import CORS
from sklearn.cluster import KMeans
import numpy as np

app = Flask(__name__)
CORS(app)

@app.route('/calculate', methods=['POST'])
def calculate_route():
    data = request.json
    affected_points = data.get('affected_points')  # [[lon, lat], ...]
    num_supply_points = data.get('num_supply_points', 3)
    
    result = kmeans_location(affected_points, num_supply_points)
    return jsonify({
        'supply_points': result.tolist(),
        'message': '选址计算完成'
    })

def kmeans_location(affected_points, num_supply_points):
    """
    基于K-Means的中值模型选址算法
    目标：最小化所有需求点到最近设施的平均距离
    """
    kmeans = KMeans(
        n_clusters=num_supply_points,
        random_state=0,
        n_init=10
    ).fit(affected_points)
    
    # 聚类中心即为最优配送中心位置
    supply_points = kmeans.cluster_centers_
    print("临时物资点位置（经纬度）：")
    print(supply_points)
    return supply_points

if __name__ == '__main__':
    app.run(debug=True, port=5000)
```

function geocodeAddress(address) {

// 计算两点间距离（Haversine公式）
function haversineDistance(routePositions) {
    const R = 6371.393; // 地球平均半径（公里）
    let totalDistance = 0;
    for (let i = 0; i < routePositions.length - 1; i++) {
        let lat1 = routePositions[i][0];
        let lon1 = routePositions[i][1];
        let lat2 = routePositions[i + 1][0];
        let lon2 = routePositions[i + 1][1];
        let dLat = toRadians(lat2 - lat1);
        let dLon = toRadians(lon2 - lon1);
        const a = Math.sin(dLat / 2) ** 2
            + Math.cos(toRadians(lat1)) * Math.cos(toRadians(lat2))
            * Math.sin(dLon / 2) ** 2;
        let c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
        totalDistance += R * c;
    }
    return totalDistance;
}

function toRadians(degrees) {
    return degrees * Math.PI / 180;
}

// 3D车辆调度模拟
function simulate3DDispatch(routePositions) {
    const velocity = 1000; // 速度
    var runTime = haversineDistance(routePositions) / velocity;
    
    const start = Cesium.JulianDate.fromDate(new Date());
    const stop = Cesium.JulianDate.addSeconds(start, runTime, new Cesium.JulianDate());
    
    viewer.clock.startTime = start.clone();
    viewer.clock.stopTime = stop.clone();
    viewer.clock.currentTime = start.clone();
    viewer.clock.clockRange = Cesium.ClockRange.LOOP_STOP;
    viewer.clock.shouldAnimate = true;
    
    // 创建位置属性
    const positionProperty = new Cesium.SampledPositionProperty();
    for (let i = 0; i < routePositions.length; i++) {
        const time = Cesium.JulianDate.addSeconds(
            start,
            haversineDistance(routePositions.slice(0, i + 1)) / velocity,
            new Cesium.JulianDate()
        );
        const position = Cesium.Cartesian3.fromDegrees(
            routePositions[i][1], 
            routePositions[i][0], 
            10
        );
        positionProperty.addSample(time, position);
    }
    
    // 添加车辆实体
    const entity = viewer.entities.add({
        availability: new Cesium.TimeIntervalCollection([
            new Cesium.TimeInterval({ start: start, stop: stop })
        ]),
        id: "move",
        position: positionProperty,
        orientation: new Cesium.VelocityOrientationProperty(positionProperty),
        model: {
            uri: "models/Bus.glb",
            minimumPixelSize: 5,
            scale: 2
        },
        path: {
            resolution: 1,
            leadTime: 0,
            width: 20,
            material: Cesium.Color.RED
        }
    });
}
