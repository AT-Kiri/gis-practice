# 三维洪水模拟模块 - 超详细实现指南

## 一、模块功能概述

三维洪水模拟模块是整个应急物资调度系统的**可视化入口**，主要功能包括：

1. **三维地球展示**：基于 Cesium 渲染三维地球，展示地形和卫星底图
2. **地区定位**：输入地区名称，视角自动飞移到目标区域
3. **洪水淹没模拟**：输入淹没高度和速度，动态展示洪水淹没过程
4. **淹没时序可视化**：用红→黄→绿→蓝渐变色彩表示淹没的时间顺序和深度
5. **灾情统计**：淹没完成后自动统计被淹没的房屋和道路数量
6. **受灾人员标记**：在地图上手动标记受灾人员位置

## 三、核心原理详解

### 3.1 洪水淹没效果原理

Cesium 的 `HypsometricSetting`（分层设色）是实现洪水淹没效果的核心：

```
原理：根据地形高度值设置颜色透明度

地形高度 0m ────────────────────────────── 完全淹没（红色）
      ↑
      │  随着水位上升，低于水位的区域逐渐显示颜色
      │
      │  高于水位的区域保持透明（显示原始地形）
      ↓
地形高度 max ───────────────────────────── 未淹没（透明）
```

**关键参数：**
- `CoverageArea`：淹没区域范围（多边形坐标）
- `MinVisibleValue`：最小可见高度（0）
- `MaxVisibleValue`：最大可见高度（当前水位）
- `ColorTable`：高度→颜色的映射表（红→黄→绿→蓝）

### 3.2 动画时钟原理

```
Cesium.Clock 控制时间轴：

startTime ────── currentTime ────── stopTime
    │                │                 │
    ▼                ▼                 ▼
  开始时刻         当前时刻           结束时刻
                   │
                   └── 每帧更新水位高度
                       currentHeight = elapsedTime × floodSpeed
```

### 3.3 淹没统计原理

```
统计逻辑：
1. 获取所有建筑/道路实体
2. 遍历每个实体，获取其位置坐标
3. 判断坐标是否在淹没区域内（矩形包围盒检测）
4. 统计在区域内的实体数量
```

/**
 * 初始化分层设色对象（洪水效果）
 */
function initHypsometricSetting() {
    hypFlood = new Cesium.HypsometricSetting();
    
    // 设置颜色映射表：高度 -> 颜色
    // 红色（低/早淹没）→ 黄色 → 绿色 → 蓝色（高/晚淹没）
    const colorTable = new Cesium.ColorTable();
    colorTable.insert(0, new Cesium.Color(1.0, 0.0, 0.0, 0.8));      // 红色
    colorTable.insert(0.25, new Cesium.Color(1.0, 0.65, 0.0, 0.7));  // 橙色
    colorTable.insert(0.5, new Cesium.Color(1.0, 1.0, 0.0, 0.6));    // 黄色
    colorTable.insert(0.75, new Cesium.Color(0.0, 1.0, 0.0, 0.5));   // 绿色
    colorTable.insert(1.0, new Cesium.Color(0.0, 0.0, 1.0, 0.4));    // 蓝色
    
    hypFlood.ColorTable = colorTable;
    hypFlood.DisplayMode = Cesium.HypsometricSettingEnum.DisplayMode.FACE;
    hypFlood.Opacity = 0.6;
    hypFlood.MinVisibleValue = 0;
    hypFlood.MaxVisibleValue = 0;
}

/**
 * 查找地区坐标

function findLocation(name) {
    // 精确匹配
    if (PRESET_LOCATIONS[name]) {
        return PRESET_LOCATIONS[name];
    }
    
    // 模糊匹配
    for (let key in PRESET_LOCATIONS) {
        if (key.includes(name) || name.includes(key)) {
            return PRESET_LOCATIONS[key];
        }
    }
    
    return null;
}

/**
 * 加载GeoJSON数据
 * @param {string} url - GeoJSON文件URL
 * @param {Cesium.Color} color - 显示颜色
 * @param {string} type - 数据类型（building/road）
 */
function loadGeoJSON(url, color, type) {
    return new Promise((resolve, reject) => {
        Cesium.Resource.fetchJson(url)
            .then(geojsonData => {
                return Cesium.GeoJsonDataSource.load(geojsonData, {
                    clampToGround: true,  // 贴地显示
                    stroke: color,
                    fill: color.withAlpha(0.5),
                    strokeWidth: 3
                });
            })
            .then(dataSource => {
                viewer.dataSources.add(dataSource);
                
                // 获取实体并设置样式
                const entities = dataSource.entities.values;
                
                entities.forEach(entity => {
                    if (entity.polygon) {
                        entity.polygon.material = color.withAlpha(0.6);
                        entity.polygon.outline = true;
                        entity.polygon.outlineColor = Cesium.Color.WHITE;
                        entity.polygon.outlineWidth = 1;
                    }
                    if (entity.polyline) {
                        entity.polyline.material = color;
                        entity.polyline.width = 2;
                    }
                });
                
                // 保存实体引用
                if (type === 'building') {
                    savedEntities_buildings = entities;
                } else if (type === 'road') {
                    savedEntities_roads = entities;
                }
                
                resolve();
            })
            .catch(error => {
                reject(error);
            });
    });
}

/**
 * 开始洪水模拟
 */
function startFlood() {
    // 获取参数
    targetFloodHeight = parseFloat(document.getElementById('floodHeight').value);
    floodSpeed = parseFloat(document.getElementById('floodSpeed').value);
    
    if (targetFloodHeight <= 0 || floodSpeed <= 0) {
        alert('请输入有效的淹没高度和速度');
        return;
    }
    
    // 设置淹没区域
    hypFlood.CoverageArea = floodAreaPositions;
    hypFlood.MinVisibleValue = 0;
    hypFlood.MaxVisibleValue = 0;  // 从0开始
    
    // 应用分层设色
    viewer.scene.globe.HypsometricSetting = {
        hypsometricSetting: hypFlood,
        analysisMode: Cesium.HypsometricSettingEnum.AnalysisRegionMode.ARM_REGION
    };
    
    // 设置时钟
    const startTime = viewer.clock.currentTime;
    const endTime = Cesium.JulianDate.addSeconds(
        startTime,
        targetFloodHeight / floodSpeed,
        new Cesium.JulianDate()
    );
    
    viewer.clock.startTime = startTime.clone();
    viewer.clock.stopTime = endTime.clone();
    viewer.clock.currentTime = startTime.clone();
    viewer.clock.clockRange = Cesium.ClockRange.CLAMPED;
    viewer.clock.shouldAnimate = true;
    
    isPaused = false;
    
    // 移除旧的监听器
    if (updateFloodCallback) {
        viewer.clock.onTick.removeEventListener(updateFloodCallback);
    }
    
    // 定义更新函数
    updateFloodCallback = function() {
        if (isPaused) return;
        
        const currentTime = viewer.clock.currentTime;
        const seconds = Cesium.JulianDate.secondsDifference(currentTime, startTime);
        
        // 计算当前水位
        currentFloodHeight = Math.min(seconds * floodSpeed, targetFloodHeight);
        hypFlood.MaxVisibleValue = currentFloodHeight;
        
        // 更新显示
        viewer.scene.globe.HypsometricSetting = {
            hypsometricSetting: hypFlood,
            analysisMode: Cesium.HypsometricSettingEnum.AnalysisRegionMode.ARM_REGION
        };
        
        // 更新UI
        document.getElementById('simulatedTime').textContent = seconds.toFixed(2) + ' 秒';
        document.getElementById('currentWaterLevel').textContent = currentFloodHeight.toFixed(2) + ' 米';
        
        // 检查是否完成
        if (currentFloodHeight >= targetFloodHeight) {
            viewer.clock.onTick.removeEventListener(updateFloodCallback);
            viewer.clock.shouldAnimate = false;
            
            // 统计淹没数据
            const submergedHouses = countSubmergedEntities(savedEntities_buildings);
            const submergedRoads = countSubmergedEntities(savedEntities_roads);
            
            document.getElementById('houseCount').textContent = submergedHouses;
            document.getElementById('roadCount').textContent = submergedRoads;
            
            alert(`洪水模拟完成！\n淹没房屋：${submergedHouses}\n淹没道路：${submergedRoads}`);
        }
    };
    
    // 添加时钟监听器
    viewer.clock.onTick.addEventListener(updateFloodCallback);
    
    console.log('洪水模拟开始');
}

/**
 * 暂停/继续洪水模拟
 */
function pauseFlood() {
    isPaused = !isPaused;
    
    if (isPaused) {
        viewer.clock.shouldAnimate = false;
        console.log('洪水模拟暂停');
    } else {
        viewer.clock.shouldAnimate = true;
        console.log('洪水模拟继续');
    }
}

/**
 * 统计淹没范围内的实体数量
 */
function countSubmergedEntities(entities) {
    if (!entities || entities.length === 0) return 0;
    
    // 获取淹没区域的矩形边界
    const rectangle = getFloodAreaRectangle();
    
    let count = 0;
    
    entities.forEach(entity => {
        let position = null;
        
        // 获取实体位置（建筑取第一个顶点，道路取中点）
        if (entity.polygon) {
            const hierarchy = entity.polygon.hierarchy.getValue(Cesium.JulianDate.now());
            if (hierarchy.positions && hierarchy.positions.length > 0) {
                position = hierarchy.positions[0];
            }
        } else if (entity.polyline) {
            const positions = entity.polyline.positions.getValue(Cesium.JulianDate.now());
            if (positions && positions.length > 0) {
                position = positions[Math.floor(positions.length / 2)];
            }
        }
        
        if (position) {
            const cartographic = Cesium.Cartographic.fromCartesian(position);
            
            // 检查是否在淹没区域内
            if (isPointInRectangle(cartographic, rectangle)) {
                count++;
            }
        }
    });
    
    return count;
}

/**
 * 获取淹没区域的矩形边界
 */
function getFloodAreaRectangle() {
    if (!floodAreaPositions || floodAreaPositions.length === 0) {
        return null;
    }
    
    let west = Infinity, south = Infinity;
    let east = -Infinity, north = -Infinity;
    
    floodAreaPositions.forEach(cartesian => {
        const cartographic = Cesium.Cartographic.fromCartesian(cartesian);
        const lon = Cesium.Math.toDegrees(cartographic.longitude);
        const lat = Cesium.Math.toDegrees(cartographic.latitude);
        
        west = Math.min(west, lon);
        south = Math.min(south, lat);
        east = Math.max(east, lon);
        north = Math.max(north, lat);
    });
    
    return { west, south, east, north };
}

/**
 * 判断点是否在矩形内
 */
function isPointInRectangle(cartographic, rectangle) {
    if (!rectangle) return false;
    
    const lon = Cesium.Math.toDegrees(cartographic.longitude);
    const lat = Cesium.Math.toDegrees(cartographic.latitude);
    
    return lon >= rectangle.west && lon <= rectangle.east &&
           lat >= rectangle.south && lat <= rectangle.north;
}

## 八、常见问题

**Q1: 地形加载慢或失败？**
- 检查网络连接
- 申请 Cesium Ion Token 替换默认地形

**Q2: GeoJSON数据不显示？**
- 检查文件路径是否正确
- 检查GeoJSON格式是否标准
- 查看浏览器控制台错误信息

**Q3: 淹没效果不明显？**
- 调整 `Opacity` 透明度
- 调整颜色映射表
- 确保地形数据已加载

**Q4: 标记图标不显示？**
- 检查 `assets/position.png` 是否存在
- 检查路径是否正确
